#include <bits/stdc++.h>
using namespace std;
#define ll long long
int N=1e6+1;
vector<ll>primes(N,1);
void gp(vector<ll>&primes)
{
    primes[0]=0;
    primes[0]=0;
    for(int i=2;i*i<=N;i++)
    {
        for(int j=i*i;j<=N;j+=i)
        {
            primes[j]=0;
        }
    }
}
int main() {
   gp(primes);
   ll t;
   cin>>t;
   while(t--)
   {
       ll n;
       cin>>n;
       if(primes[n]==1)
       {
           cout<<"prime"<<endl;
       }
       else
       {
           cout<<"Not prime"<<endl;
       }
   }
    return 0;
}
